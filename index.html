    private void btnBukaActionPerformed(java.awt.event.ActionEvent evt) {                                        
        // Membuat objek JFileChooser
    javax.swing.JFileChooser fc = new javax.swing.JFileChooser();
    
    // Menampilkan dialog Open File
    int returnVal = fc.showOpenDialog(this);
    
    // Mengecek apakah user menekan tombol 'Open'
    if (returnVal == javax.swing.JFileChooser.APPROVE_OPTION) {
        java.io.File file = fc.getSelectedFile();
        // Menampilkan alamat lengkap file ke textfield
        txtPath.setText(file.getAbsolutePath());
    } else {
        txtPath.setText("Batal memilih file.");
    }
    }                                       

    private void btnSimpanActionPerformed(java.awt.event.ActionEvent evt) {                                          
     // 1. Cek apakah user sudah memilih file (txtPath tidak kosong atau default)
    String path = txtPath.getText();
    
    if (path.isEmpty() || path.equals("Batal memilih file.")) {
        // Jika belum pilih file, tidak terjadi apa-apa (diam) atau beri peringatan
        // javax.swing.JOptionPane.showMessageDialog(this, "Silahkan pilih file terlebih dahulu!");
        return; 
    }

    // 2. Jika sudah ada file, baru munculkan dialog simpan
    javax.swing.JFileChooser fc = new javax.swing.JFileChooser();
    
    // Set nama file default sesuai dengan file yang dipilih sebelumnya
    java.io.File fileAsal = new java.io.File(path);
    fc.setSelectedFile(new java.io.File("copy_of_" + fileAsal.getName()));

    int returnVal = fc.showSaveDialog(this);
    
    if (returnVal == javax.swing.JFileChooser.APPROVE_OPTION) {
        java.io.File fileTujuan = fc.getSelectedFile();
        javax.swing.JOptionPane.showMessageDialog(this, "File berhasil diproses ke: " + fileTujuan.getAbsolutePath());
    }
    }
